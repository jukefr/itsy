#!/usr/bin/env python3
"""Merge user data from three sources with conflict resolution."""
import json, pandas as pd

FIELD_MAPS = {
    "source_a": {"id":"user_id","full_name":"name","email":"email","registration_date":"created_date","status":"status"},
    "source_b": {"user_id":"user_id","name":"name","email_address":"email","created_at":"created_date","is_active":"status"},
    "source_c": {"userId":"user_id","userName":"name","email":"email","joined":"created_date","active":"status"},
}

def load_source_a():
    with open("/data/source_a/users.json") as f:
        data = json.load(f)
    return pd.DataFrame(data).rename(columns=FIELD_MAPS["source_a"])

def load_source_b():
    df = pd.read_csv("/data/source_b/users.csv").rename(columns=FIELD_MAPS["source_b"])
    df["status"] = df["status"].map({"true":"active","True":"active","false":"inactive","False":"inactive"})
    return df

def load_source_c():
    df = pd.read_parquet("/data/source_c/users.parquet").rename(columns=FIELD_MAPS["source_c"])
    df["status"] = df["status"].map({True:"active",False:"inactive"})
    return df

def detect_conflicts(df_a, df_b, df_c):
    conflicts = []
    all_ids = sorted(set(df_a["user_id"].tolist() + df_b["user_id"].tolist() + df_c["user_id"].tolist()))
    for uid in all_ids:
        row_a = df_a[df_a["user_id"] == uid]
        row_b = df_b[df_b["user_id"] == uid]
        row_c = df_c[df_c["user_id"] == uid]
        for field in ["name", "email", "created_date"]:
            vals = {}
            if not row_a.empty:
                v = row_a.iloc[0][field]
                if pd.notna(v): vals["source_a"] = str(v)
            if not row_b.empty:
                v = row_b.iloc[0][field]
                if pd.notna(v): vals["source_b"] = str(v)
            if not row_c.empty:
                v = row_c.iloc[0][field]
                if pd.notna(v): vals["source_c"] = str(v)
            unique = set(vals.values())
            if len(unique) > 1:
                conflicts.append({"user_id": int(uid), "field": field, "values": vals,
                    "selected": vals.get("source_a", vals.get("source_b", vals.get("source_c")))})
    return conflicts

def main():
    df_a, df_b, df_c = load_source_a(), load_source_b(), load_source_c()
    conflicts = detect_conflicts(df_a, df_b, df_c)
    merged = df_a.copy()
    new_b = df_b[~df_b["user_id"].isin(merged["user_id"])]
    merged = pd.concat([merged, new_b], ignore_index=True)
    new_c = df_c[~df_c["user_id"].isin(merged["user_id"])]
    merged = pd.concat([merged, new_c], ignore_index=True)
    merged["user_id"] = merged["user_id"].astype(int)
    merged["created_date"] = merged["created_date"].apply(lambda x: str(x).split()[0] if isinstance(x, pd.Timestamp) else str(x))
    merged = merged[["user_id","name","email","created_date","status"]]
    merged.to_parquet("/app/merged_users.parquet", index=False)
    with open("/app/conflicts.json","w") as f:
        json.dump({"total_conflicts": len(conflicts), "conflicts": conflicts}, f, indent=2)
    print(f"Merged {len(merged)} users, {len(conflicts)} conflicts")

if __name__ == "__main__":
    main()
