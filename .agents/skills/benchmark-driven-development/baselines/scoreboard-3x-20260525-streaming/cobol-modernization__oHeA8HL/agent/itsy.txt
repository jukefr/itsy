"""Re-implementation of /app/src/program.cbl (COBOL BookForum program)"""

ACCOUNTS_FILE = "data/ACCOUNTS.DAT"
BOOKS_FILE = "data/BOOKS.DAT"
TRANSACTIONS_FILE = "data/TRANSACTIONS.DAT"
INPUT_FILE = "src/INPUT.DAT"

ACCOUNT_RECORD_SIZE = 34
BOOK_RECORD_SIZE = 32
TRANSACTION_RECORD_SIZE = 22
INPUT_RECORD_SIZE = 22


def read_input():
    with open(INPUT_FILE, "rb") as f:
        data = f.read()
    if len(data) < INPUT_RECORD_SIZE:
        data = data + b' ' * (INPUT_RECORD_SIZE - len(data))
    buyer_id = data[0:4].decode('ascii')
    seller_id = data[4:8].decode('ascii')
    book_id = data[8:12].decode('ascii')
    amount_str = data[12:22].decode('ascii')
    amount = int(amount_str.replace(' ', '0'))
    return buyer_id, seller_id, book_id, amount


def read_accounts():
    accounts = []
    with open(ACCOUNTS_FILE, "rb") as f:
        data = f.read()
    pos = 0
    while pos + ACCOUNT_RECORD_SIZE <= len(data):
        rec = data[pos:pos + ACCOUNT_RECORD_SIZE]
        accounts.append({
            'id': rec[0:4].decode('ascii'),
            'name': rec[4:24].decode('ascii'),
            'balance': int(rec[24:34].decode('ascii'))
        })
        pos += ACCOUNT_RECORD_SIZE
    return accounts


def write_accounts(accounts):
    with open(ACCOUNTS_FILE, "wb") as f:
        for acc in accounts:
            f.write(acc['id'].encode('ascii') +
                    acc['name'].encode('ascii') +
                    str(acc['balance']).zfill(10).encode('ascii'))


def read_books():
    books = []
    with open(BOOKS_FILE, "rb") as f:
        data = f.read()
    pos = 0
    while pos + BOOK_RECORD_SIZE <= len(data):
        rec = data[pos:pos + BOOK_RECORD_SIZE]
        books.append({
            'id': rec[0:4].decode('ascii'),
            'title': rec[4:24].decode('ascii'),
            'owner': rec[24:28].decode('ascii')
        })
        pos += BOOK_RECORD_SIZE
    return books


def write_books(books):
    with open(BOOKS_FILE, "wb") as f:
        for book in books:
            f.write(book['id'].encode('ascii') +
                    book['title'].encode('ascii') +
                    book['owner'].encode('ascii'))


def append_transaction(book_id, amount, seller_id, buyer_id):
    with open(TRANSACTIONS_FILE, "ab") as f:
        f.write(book_id.encode('ascii') +
                str(amount).zfill(10).encode('ascii') +
                seller_id.encode('ascii') +
                buyer_id.encode('ascii'))


def validate_users_and_book(buyer_id, seller_id, book_id):
    accounts = read_accounts()
    books = read_books()
    buyer_found = any(a['id'] == buyer_id for a in accounts)
    seller_found = any(a['id'] == seller_id for a in accounts)
    book_found = False
    valid_owner = False
    for b in books:
        if b['id'] == book_id:
            book_found = True
            if b['owner'] == seller_id:
                valid_owner = True
    return buyer_found, seller_found, book_found, valid_owner


def process_transaction(buyer_id, seller_id, book_id, amount):
    accounts = read_accounts()
    for acc in accounts:
        if acc['id'] == buyer_id:
            acc['balance'] -= amount
        if acc['id'] == seller_id:
            acc['balance'] += amount
    write_accounts(accounts)

    books = read_books()
    for book in books:
        if book['id'] == book_id:
            book['owner'] = buyer_id
    write_books(books)

    append_transaction(book_id, amount, seller_id, buyer_id)


def main():
    buyer_id, seller_id, book_id, amount = read_input()
    buyer_found, seller_found, book_found, valid_owner = validate_users_and_book(
        buyer_id, seller_id, book_id)
    if buyer_found and seller_found and book_found and valid_owner:
        process_transaction(buyer_id, seller_id, book_id, amount)


if __name__ == "__main__":
    main()
