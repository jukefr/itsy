    book_id = data[8:12]
    amount_str = data[12:22].replace(" ", "0")
    try:

    except ValueError:
        amount = 0
    return buyer_id, seller_id, book_id, amount


def read_accounts():
    accounts = []
    with open(ACCOUNTS_FILE, "r") as f:
        data = f.read()
    idx = 0
    while idx + ACCOUNT_RECORD_SIZE <= len(data):
        record = data[idx : idx + ACCOUNT_RECORD_SIZE]
        account_id = record[0:4]
        account_name = record[4:24]
        account_balance = int(record[24:34])
        accounts.append((account_id, account_name, account_balance))
        idx += ACCOUNT_RECORD_SIZE
    return accounts


def write_accounts(accounts):
    with open(ACCOUNTS_FILE, "w") as f:
        for account_id, account_name, account_balance in accounts:
            record = account_id + account_name + f"{account_balance:010d}"
            f.write(record)

    books = []

        book_id = record[0:4]
        book_title = record[4:24]

        books.append((book_id, book_title, book_owner))
