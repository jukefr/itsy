"""Merge user data from three sources with conflict resolution."""
import json
import pandas as pd
import pyarrow.parquet as pq

with open('/data/source_a/users.json', 'r') as f:

# --- 2. Normalize field names ---
def normalize_source_a(data):
    records = []
    for row in data:
        records.append({
            'user_id': int(row['id']),
            'name': row['full_name'],
            'email': row['email'],
            'created_date': row['registration_date'],
            'status': row.get('status', None)
        })
    return records

def normalize_source_b(df):
    df = df.copy()
    df.columns = [c.lower() for c in df.columns]
    records = []
    for _, row in df.iterrows():
        is_active = row.get('is_active')
        status_val = str(is_active).lower() if pd.notna(is_active) else None
        records.append({
            'user_id': int(row['user_id']),
            'name': str(row['name']),
            'email': str(row['email_address']),
            'created_date': str(row['created_at']),
            'status': status_val
        })
    return records

def normalize_source_c(df):
    df = df.copy()
    records = []
    for _, row in df.iterrows():
        active = row.get('active')
        status_val = str(active).lower() if pd.notna(active) else None
        records.append({
            'user_id': int(row['userId']),
            'name': str(row['userName']),
            'email': str(row['email']),
            'created_date': str(row['joined']),
            'status': status_val
        })
    return records

norm_a = normalize_source_a(source_a)
norm_b = normalize_source_b(source_b_df)
norm_c = normalize_source_c(source_c_df)

# --- 3. Build unified dataframe and detect conflicts ---
a_dict = {r['user_id']: r for r in norm_a}
b_dict = {r['user_id']: r for r in norm_b}
c_dict = {r['user_id']: r for r in norm_c}
all_ids = sorted(set(a_dict.keys()) | set(b_dict.keys()) | set(c_dict.keys()))

merged_rows = []
conflicts = []

for uid in all_ids:
    row_a = a_dict.get(uid)
    row_b = b_dict.get(uid)
    row_c = c_dict.get(uid)
    selected = {}

        val_a = row_a[field] if row_a else None
        val_b = row_b[field] if row_b else None
        val_c = row_c[field] if row_c else None
        selected[field] = val_a if val_a is not None else (val_b if val_b is not None else val_c)
    merged_rows.append(selected)
    fields_to_check = ['name', 'email', 'created_date', 'status']
    for field in fields_to_check:
        vals = {}

            vals['source_a'] = row_a[field]
        if row_b and row_b[field] is not None:

        unique_vals = set(str(v) for v in vals.values())
        if len(unique_vals) > 1:
            conflicts.append({
                'user_id': uid,
                'field': field,
                'values': vals,
