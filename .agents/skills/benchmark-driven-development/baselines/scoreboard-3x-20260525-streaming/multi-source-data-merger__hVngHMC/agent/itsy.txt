from datetime import datetime

# Field mappings

    'id': 'user_id',
    'userId': 'user_id',
    'user_id': 'user_id',
    'full_name': 'name',
    'name': 'name',
    'userName': 'name',

    'registration_date': 'created_date',
    'created_at': 'created_date',
    'joined': 'created_date',

STATUS_MAP = {
    'status': 'status',
    'is_active': 'status',
    'active': 'status',
}

def normalize_status(value):
    if isinstance(value, bool):
        return 'active' if value else 'inactive'
    if isinstance(value, str):
        val = value.lower()
        if val in ('true', 'active', 'yes'):
            return 'active'
        elif val in ('false', 'inactive', 'no'):
            return 'inactive'
    return str(value)

def normalize_date(date_str):
    try:
        dt = datetime.strptime(date_str, '%Y-%m-%d')
        return dt.strftime('%Y-%m-%d')
    except ValueError:
        try:
            dt = datetime.strptime(date_str, '%Y/%m/%d')
            return dt.strftime('%Y-%m-%d')
        except ValueError:
            return date_str

            rec[FIELD_MAP[k]] = v
